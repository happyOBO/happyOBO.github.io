```
var jbRandom = Math.random();
var jbRandom = Math.floor(jbRandom * 10 ); 
var hello = "안녕하세요";
var nothello = "아니..안녕하세요";
if(jbRandom < 5)
{
    document.body.innerText = hello;
    
};
if(jbRandom >= 5)
{
    document.body.innerText = nothello;
};
console.log(jbRandom);
/* 자바스크립트 콘솔에 랜덤변수 값 출력 */
```
